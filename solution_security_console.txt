import React, { useState, useEffect } from 'react';
import { Search, LogOut, Clock, User, Building2 } from 'lucide-react';
import api from '../api';

const SecurityConsole = () => {
  const [activeVisitors, setActiveVisitors] = useState([]);
  const [loading, setLoading] = useState(true);
  const [searchTerm, setSearchTerm] = useState('');

  useEffect(() => {
    fetchActiveVisitors();
  }, []);

  const fetchActiveVisitors = async () => {
    try {
      const response = await api.get('/admin/active-visitors');
      setActiveVisitors(response.data);
      setLoading(false);
    } catch (error) {
      console.error('Error fetching active visitors:', error);
      setLoading(false);
    }
  };

  const handleCheckout = async (approvalId) => {
    if (window.confirm('Are you sure you want to check out this visitor?')) {
      try {
        await api.patch(`/visitors/${approvalId}/checkout`);
        // Refresh the list
        fetchActiveVisitors();
        alert('Visitor checked out successfully');
      } catch (error) {
        console.error('Error checking out visitor:', error);
        alert('Failed to check out visitor: ' + (error.response?.data?.message || error.message));
      }
    }
  };

  const filteredVisitors = activeVisitors.filter(approval => 
    approval.visitorName?.toLowerCase().includes(searchTerm.toLowerCase()) ||
    approval.visitor?.email?.toLowerCase().includes(searchTerm.toLowerCase()) ||
    approval.visitor?.phone?.includes(searchTerm)
  );

  return (
    <div className="p-6">
      <div className="mb-8">
        <h1 className="text-2xl font-bold text-gray-900">Security Console</h1>
        <p className="text-gray-600">Manage Visitor Check-In and Check-Out</p>
      </div>

      {/* Active Visitors Section */}
      <div className="bg-white rounded-xl shadow-sm border border-gray-200 overflow-hidden">
        <div className="p-6 border-b border-gray-200 flex justify-between items-center">
          <div className="flex items-center gap-2">
            <div className="p-2 bg-green-100 rounded-lg">
              <User className="w-5 h-5 text-green-600" />
            </div>
            <h2 className="text-lg font-semibold text-gray-900">Active Visitors</h2>
            <span className="px-2.5 py-0.5 bg-green-100 text-green-700 rounded-full text-sm font-medium">
              {activeVisitors.length}
            </span>
          </div>
          
          <div className="relative">
            <Search className="w-5 h-5 text-gray-400 absolute left-3 top-1/2 -translate-y-1/2" />
            <input
              type="text"
              placeholder="Search visitors..."
              value={searchTerm}
              onChange={(e) => setSearchTerm(e.target.value)}
              className="pl-10 pr-4 py-2 border border-gray-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-blue-500 w-64"
            />
          </div>
        </div>

        <div className="overflow-x-auto">
          <table className="w-full">
            <thead className="bg-gray-50">
              <tr>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Visitor</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Host & Company</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Check-In Time</th>
                <th className="px-6 py-3 text-left text-xs font-medium text-gray-500 uppercase tracking-wider">Status</th>
                <th className="px-6 py-3 text-right text-xs font-medium text-gray-500 uppercase tracking-wider">Action</th>
              </tr>
            </thead>
            <tbody className="divide-y divide-gray-200">
              {loading ? (
                <tr>
                  <td colSpan="5" className="px-6 py-8 text-center text-gray-500">
                    Loading active visitors...
                  </td>
                </tr>
              ) : filteredVisitors.length === 0 ? (
                <tr>
                  <td colSpan="5" className="px-6 py-8 text-center text-gray-500">
                    No active visitors found
                  </td>
                </tr>
              ) : (
                filteredVisitors.map((approval) => (
                  <tr key={approval.id} className="hover:bg-gray-50 transition-colors">
                    <td className="px-6 py-4">
                      <div className="flex items-center">
                        <div className="h-10 w-10 flex-shrink-0">
                          {approval.visitor?.imagePath ? (
                             <img 
                               src={`http://localhost:8080/visitor-photos/${approval.visitor.imagePath.split('\\').pop().split('/').pop()}`} 
                               alt={approval.visitorName}
                               className="h-10 w-10 rounded-full object-cover border border-gray-200"
                               onError={(e) => {
                                 e.target.onerror = null; 
                                 e.target.src = 'https://via.placeholder.com/40';
                               }}
                             />
                          ) : (
                            <div className="h-10 w-10 rounded-full bg-gray-200 flex items-center justify-center">
                              <User className="w-6 h-6 text-gray-400" />
                            </div>
                          )}
                        </div>
                        <div className="ml-4">
                          <div className="text-sm font-medium text-gray-900">{approval.visitorName}</div>
                          <div className="text-sm text-gray-500">{approval.visitor?.visitorType}</div>
                        </div>
                      </div>
                    </td>
                    <td className="px-6 py-4">
                      <div className="flex flex-col">
                        <span className="text-sm font-medium text-gray-900">
                          {approval.host?.name || 'Unknown Host'}
                        </span>
                        <div className="flex items-center text-sm text-gray-500 mt-1">
                          <Building2 className="w-3 h-3 mr-1" />
                          {approval.company?.name || 'Unknown Company'}
                        </div>
                      </div>
                    </td>
                    <td className="px-6 py-4">
                      <div className="flex items-center text-sm text-gray-500">
                        <Clock className="w-4 h-4 mr-1.5" />
                        {new Date(approval.inTime).toLocaleString()}
                      </div>
                    </td>
                    <td className="px-6 py-4">
                      <span className="px-2.5 py-0.5 rounded-full text-xs font-medium bg-green-100 text-green-800 border border-green-200">
                        Checked In
                      </span>
                    </td>
                    <td className="px-6 py-4 text-right">
                      <button
                        onClick={() => handleCheckout(approval.id)}
                        className="inline-flex items-center px-3 py-1.5 border border-red-200 text-sm font-medium rounded-lg text-red-700 bg-red-50 hover:bg-red-100 hover:border-red-300 transition-colors focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-red-500"
                      >
                        <LogOut className="w-4 h-4 mr-1.5" />
                        Check Out
                      </button>
                    </td>
                  </tr>
                ))
              )}
            </tbody>
          </table>
        </div>
      </div>
    </div>
  );
};

export default SecurityConsole;
